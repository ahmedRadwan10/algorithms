Insertion Sort

        
    function insertionSort(arr) {
        for (let i = 1; i < arr.length; i++){
            let currentValue = arr[i];
            for (let j = i - 1; j >= 0 && arr[j] > currentValue; j--){
                if (currentValue < arr[j]) {
                    let temp = arr[j];
                    arr[j] = currentValue;
                    arr[j + 1] = temp;
                }
            }
        }
        return arr;
    }
    console.log(insertionSort([11,1,7,12,15,45,81,20,22,66,32]))
